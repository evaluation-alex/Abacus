Note: Due to the large number of combinatorial samples,
Abacus combinatorics use an Iterator pattern to succesively and consistently
generate all combinatorial objects without storing all of them in memory at once






Abacus.Tensors
---
t = Abacus.Tensor([["a1"],["b1","b2"],["c1","c2","c3"]])
t.total()
6
t.next()
[ 'a1', 'b1', 'c1' ]
t.hasNext()
true
t.next()
[ 'a1', 'b1', 'c2' ]
[ [ 'a1', 'b1', 'c1' ], 0, [ 'a1', 'b1', 'c1' ] ]
[ [ 'a1', 'b1', 'c2' ], 1, [ 'a1', 'b1', 'c2' ] ]
[ [ 'a1', 'b1', 'c3' ], 2, [ 'a1', 'b1', 'c3' ] ]
[ [ 'a1', 'b2', 'c1' ], 3, [ 'a1', 'b2', 'c1' ] ]
[ [ 'a1', 'b2', 'c2' ], 4, [ 'a1', 'b2', 'c2' ] ]
[ [ 'a1', 'b2', 'c3' ], 5, [ 'a1', 'b2', 'c3' ] ]
t.forward()
while (t.hasPrev()) echo(t.prev())
[ [ 'a1', 'b2', 'c3' ], 5 ]
[ [ 'a1', 'b2', 'c2' ], 4 ]
[ [ 'a1', 'b2', 'c1' ], 3 ]
[ [ 'a1', 'b1', 'c3' ], 2 ]
[ [ 'a1', 'b1', 'c2' ], 1 ]
[ [ 'a1', 'b1', 'c1' ], 0 ]
t.random()
[ 'a1', 'b2', 'c2' ]
get tensors in unique random order
t.randomise()
while(t.hasRandomNext()) echo(t.randomNext())
[ 'a1', 'b1', 'c1' ]
[ 'a1', 'b2', 'c3' ]
[ 'a1', 'b2', 'c1' ]
[ 'a1', 'b2', 'c2' ]
[ 'a1', 'b1', 'c3' ]
[ 'a1', 'b1', 'c2' ]
t.dispose()
t = Abacus.Tensor([["a1"],["b1","b2"],["c1","c2","c3"]])
get just last 3 tensors
t.range(-3,-1)
[ [ 'a1', 'b2', 'c1' ],
  [ 'a1', 'b2', 'c2' ],
  [ 'a1', 'b2', 'c3' ] ]
get just last 3 tensors in reverse order
t.range(-1,-3)
[ [ 'a1', 'b2', 'c3' ],
  [ 'a1', 'b2', 'c2' ],
  [ 'a1', 'b2', 'c1' ] ]
